CREATING TABLE :-
==================
1. PHARMECY:

 CREATE TABLE PHARMECY( p_id number(3) PRIMARY KEY, p_name varchar2(15) NOT NULL, p_location1 varchar2(15) NOT NULL, p_location2 varchar2(15) NOT NULL,
 p_location3 varchar2(15) NOT NULL);

2. FOREIGNKEYTABLE:
 CREATE TABLE FOREIGNKEYTABLE( p_id number(3), m_id number(3), constraint  PK_pm_id PRIMARY KEY (p_id , m_id));

3. DOCTOR : 

 CREATE TABLE DOCTOR ( d_id number(3) PRIMARY KEY, d_name varchar2(15) NOT NULL, d_speciality1 varchar2(15) NOT NULL, d_speciality2 varchar2(15) NOT NULL, 
 d_speciality3 varchar2(15) NOT NULL,ad_id number(3) NOT Null);

4. ADDRESS : 

CREATE TABLE ADDRESS( ad_id number(3) PRIMARY KEY, d_city varchar2(15) NOT NULL, d_country varchar2(20) NOT NULL);

Adding ad_id as FOREIGN KEY to ADDRESS table :-

ALTER TABLE DOCTOR ADD CONSTRAINT fk_address FOREIGN KEY (ad_id) REFERENCES ADDRESS (ad_id);

5. MEDICINE :

CREATE TABLE MEDICINE (m_id number(3) PRIMARY KEY, m_name varchar2(30) NOT NULL, m_company varchar2(20) NOT NULL, d_id number(3) NOT NULL, 
CONSTRAINT FK_DoctorId FOREIGN KEY (d_id) REFERENCES DOCTOR(d_id));
