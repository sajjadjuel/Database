1. CREATE TABLE PUBLISHERS (pubid NUMBER(10) PRIMARY KEY, pname VARCHAR2(15), email VARCHAR2(10), phone NUMBER(20));

2. CREATE TABLE SUBJECTS (subid NUMBER(10) PRIMARY KEY, sname VARCHAR2(15));

3. CREATE TABLE TITLES( title_id NUMBER(10) PRIMARY KEY, title VARCHAR2(15), pub_date DATE, cover VARCHAR2(10), price NUMBER(6,4), pubid NUMBER(20), subid NUMBER(10),
                    CONSTRAINT FK1 FOREIGN KEY (pubid)  REFERENCES PUBLISHERS (pubid), CONSTRAINT FK2 FOREIGN KEY (subid)  REFERENCES SUBJECTS (subid));

4. CREATE TABLE AUTHORS (auid NUMBER(10) PRIMARY KEY, aname VARCHAR2(20), email VARCHAR2(30), phone NUMBER(10));

5. CREATE TABLE TITLEAUTHORS ( importance VARCHAR2(90), title_id NUMBER(10),  CONSTRAINT FK3 FOREIGN KEY ( title_id)  REFERENCES TITLES ( title_id),
                     auth_id NUMBER(10), CONSTRAINT FK4 FOREIGN KEY ( auth_id)  REFERENCES AUTHORS ( auid));
